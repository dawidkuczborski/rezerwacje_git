      year: monthRef.getFullYear(),
      month: monthRef.getMonth() + 1,

    console.log("‚úÖ Dostƒôpne dni:", available);
    setAvailableDays(available);
  } catch (err) {
    console.error("‚ùå B≈ÇƒÖd pobierania dni:", err);
    setAvailableDays([]);
  }

console.log("‚úÖ Us≈Çugi przypisane do pracownik√≥w:", filtered.map(f => ({

  name: f.name,

setAvailableServices(filtered);




          }

        }

		// üîß Fix: dodaj id, je≈õli istnieje tylko service_id
if (stored.service_id && !stored.id) {

// ‚úÖ Auto-przej≈õcie do najbli≈ºszego dnia z wolnymi terminami (bez pƒôtli)
useEffect(() => {
  if (!availableDays.length) return;

  const today = new Date();
  today.setHours(0, 0, 0, 0);

  if (date < today) {

  }

  // je≈õli obecny dzie≈Ñ nie jest dostƒôpny
  if (!availableDays.includes(currentIso)) {
    const nextAvailable = availableDays
      .map((d) => new Date(d))
      .find((d) => d >= today);

    // aktualizuj tylko, je≈õli to naprawdƒô inny dzie≈Ñ
    if (nextAvailable && formatIsoDate(nextAvailable) !== currentIso) {
      console.log("üìÖ Brak slot√≥w ‚Äî przeskakujƒô do:", nextAvailable);
      setDate(nextAvailable);
      setMonthDate(new Date(nextAvailable.getFullYear(), nextAvailable.getMonth(), 1));
    }
  }
}, [availableDays]); // ‚ùóÔ∏èbez 'date' w zale≈ºno≈õciach

      const storedService = JSON.parse(localStorage.getItem("selectedService"));
      if (storedService?.fromRebook) {

  if (rebookEmployeeId) {

  }

  if (!allowedEmpIds.length) return;

  setAvailableServices((prev) => {
    const filtered = prev.filter((svc) => {

      if (!svc.employee_ids || svc.employee_ids.length === 0) return false;
      // je≈õli kt√≥rykolwiek pracownik z allowedEmpIds ma tƒô us≈Çugƒô ‚Üí poka≈º
      return svc.employee_ids.some((id) => allowedEmpIds.includes(id));

      id: f.id,
      name: f.name,
      employee_ids: f.employee_ids
    })));

    return filtered;
  });
}, [employees]);
















  // refresh slots unless restoring rebook
useEffect(() => {

  if (isRestoringRebook) {

      start.setHours(h, m, 0, 0);
      const end = new Date(start.getTime() + total * 60000);
      setComputedEndTime(`${String(end.getHours()).padStart(2, "0")}:${String(end.getMinutes()).padStart(2, "0")}`);
    } else {
      setComputedEndTime(null);
    }
  }, [selectedSlot, service, selectedAddons]);

    if (selectedSlot && selectedSlot.employee_id === slot.employee_id && selectedSlot.start_time === slot.start_time) {
      setSelectedSlot(null);
    } else {
      setSelectedSlot(slot);

  setSelectedAddons((prev) => {

          employee_id: selectedSlot.employee_id,
          service_id: service.id,
          date: formatIsoDate(date),
          start_time: selectedSlot.start_time,
          end_time: computedEndTime || selectedSlot.end_time,
          addons: selectedAddons.map((a) => a.id),
        },
        token ? { headers: { Authorization: `Bearer ${token}` } } : {}
      );
      setMsg("‚úÖ Rezerwacja potwierdzona!");
    } catch (err) {
      console.error("‚ùå B≈ÇƒÖd rezerwacji:", err);
      setMsg(err.response?.data?.error || "‚ùå Nie uda≈Ço siƒô zarezerwowaƒá terminu");
    }
  };

  // calendar helpers
  const weekdays = ["PON.", "WT.", "≈öR.", "CZW.", "PT.", "SOB.", "NIED."];
  const buildMonthGrid = useCallback((d) => {
    const first = new Date(d.getFullYear(), d.getMonth(), 1);
    const last = new Date(d.getFullYear(), d.getMonth() + 1, 0);
    const startOffset = (first.getDay() + 6) % 7;
    const totalCells = startOffset + last.getDate();
    const rows = Math.ceil(totalCells / 7);
    const cells = [];
    let current = new Date(first);
    current.setDate(first.getDate() - startOffset);
    for (let r = 0; r < rows; r++) {
      const row = [];
      for (let c = 0; c < 7; c++) {
        row.push(new Date(current));
        current.setDate(current.getDate() + 1);
      }
      cells.push(row);
    }
    return cells;
  }, []);

  const prevMonth = () => setMonthDate((m) => { const n = new Date(m); n.setMonth(n.getMonth() - 1); return n; });
  const nextMonth = () => setMonthDate((m) => { const n = new Date(m); n.setMonth(n.getMonth() + 1); return n; });

  // totals
  const totalPrice = useMemo(() => Number(service?.price || 0) + selectedAddons.reduce((s, a) => s + Number(a.price || 0), 0), [service, selectedAddons]);
  const totalDuration = useMemo(() => Number(service?.duration_minutes || 0) + selectedAddons.reduce((s, a) => s + Number(a.duration_minutes || 0), 0), [service, selectedAddons]);

  if (!pageReady && !isRestoringRebook) {
    return <div className={`${pageBgClass} min-h-screen flex items-center justify-center`}>≈Åadowanie...</div>;
  }

  return (
    <div className={`${pageBgClass} min-h-screen font-inter p-[10px] sm:p-[clamp(5px,4vw,10px)]`}>
      {/* Header */}
      <div className="flex items-center gap-3 mb-3">
        <button
          onPointerDown={() => { localStorage.removeItem("lockEmployeeSelection"); navigate("/services"); }}
          aria-label="Powr√≥t"
          className={`bg-transparent border-0 text-[22px] cursor-pointer p-1 transition-transform duration-100 ${isDark ? "text-white" : "text-gray-900"}`}
        >
          ‚Üê
        </button>
        <h1 className="m-0 text-[22px] font-semibold">Um√≥w wizytƒô</h1>
        <button onClick={() => setTheme(isDark ? "light" : "dark")} className={`ml-auto bg-transparent rounded-md px-2 py-1 text-sm border ${borderClass} ${isDark ? "text-white" : "text-gray-900"} cursor-pointer`}>
          {isDark ? "‚òÄÔ∏è Jasny" : "üåô Ciemny"}
        </button>
      </div>

      {/* Top scrollable content */}
      <div className="pb-2">
        {!lockEmployeeSelection && (
          <div className="flex gap-3 overflow-x-auto pb-2 mb-3">
            <div onPointerDown={() => { setSelectedEmployee("any"); setSelectedSlot(null); }} className="min-w-[72px] text-center cursor-pointer">
              <div className={`w-16 h-16 rounded-full mx-auto mb-2 relative flex items-center justify-center`} style={{ background: isDark ? "#171717" : "#f2f2f2", border: selectedEmployee === "any" ? `3px solid ${PRIMARY}` : `3px solid ${isDark ? "#171717" : "#f2f2f2"}` }}>
                <div className="text-[26px]">üë§</div>
              </div>
              <div className={`text-[12px] ${selectedEmployee === "any" ? "" : subtextClass}`} style={{ color: selectedEmployee === "any" ? PRIMARY : undefined }}>Dowolna</div>
            </div>

            {employees.map((emp) => {
              const selected = selectedEmployee === emp.id;
              return (
                <div key={emp.id} className="min-w-[88px] text-center cursor-pointer" onPointerDown={() => { setSelectedEmployee(emp.id); setSelectedSlot(null); }}>
                  <div className="relative mx-auto mb-2 w-16 h-16">
                    <div className="absolute inset-0 rounded-full p-[3px] box-border flex items-center justify-center" style={{ background: `linear-gradient(180deg, ${PRIMARY}, ${PRIMARY})`, opacity: selected ? 1 : 0.14 }}>
                      <div className={`w-full h-full rounded-full overflow-hidden flex items-center justify-center`} style={{ background: isDark ? "#111" : "#fff" }}>
                        {emp.image_url ? <img src={`${backendBase}/uploads/${emp.image_url}`} alt={emp.name} className="w-full h-full object-cover" /> : <div className={`${isDark ? "text-white" : "text-gray-900"} text-lg`}>{emp.name ? emp.name[0].toUpperCase() : "?"}</div>}
                      </div>
                    </div>
                    {selected && (
                      <div className="absolute -right-1 -bottom-1 w-5 h-5 rounded-full flex items-center justify-center" style={{ background: PRIMARY, boxShadow: `0 2px 6px ${PRIMARY}66` }}>
                        <Check size={12} color="#fff" />
                      </div>
                    )}
                  </div>
                  <div className={`text-[12px] ${selected ? "" : subtextClass} truncate`} style={{ color: selected ? PRIMARY : undefined }}>{emp.name}</div>
                </div>
              );
            })}
          </div>
        )}

        {/* calendar header */}
        <div className="flex justify-between items-center mb-2">
          <div className="flex items-center gap-2">
            <button onClick={prevMonth} className={`bg-transparent border-0 p-1 ${isDark ? "text-white" : "text-gray-900"} cursor-pointer`} aria-label="Poprzedni miesiƒÖc"><ChevronLeft size={18} /></button>
            <div className="text-base font-semibold">{monthDate.toLocaleDateString("pl-PL", { month: "long", year: "numeric" })}</div>
            <button onClick={nextMonth} className={`bg-transparent border-0 p-1 ${isDark ? "text-white" : "text-gray-900"} cursor-pointer`} aria-label="Nastƒôpny miesiƒÖc"><ChevronRight size={18} /></button>
          </div>
          <div className={`text-sm ${subtextClass}`}>Wybierz datƒô</div>
        </div>

        {/* weekdays */}
        <div className="grid grid-cols-7 gap-2 mb-1">{weekdays.map((w) => <div key={w} className={`text-center text-sm ${subtextClass}`}>{w}</div>)}</div>

        {/* calendar grid */}
        <div className="rounded-lg p-3 mb-3 bg-transparent shadow-none">
          <div className="grid grid-cols-7 gap-2 justify-items-center" style={{ transform: window.innerWidth < 480 ? "scale(0.95)" : undefined }}>
            {buildMonthGrid(monthDate).flat().map((cellDate, idx) => {
  const isCurrentMonth = cellDate.getMonth() === monthDate.getMonth();
  const isSelected = cellDate.toDateString() === date.toDateString();
  const iso = formatIsoDate(cellDate);
  const isAvailable = availableDays.includes(iso);

  // üîí Zablokuj dni przesz≈Çe
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  const isPast = cellDate < today;

  const isClickable = isAvailable && !isPast;

  return (
    <button
      key={idx}
      onPointerDown={() => {
        if (!isClickable) return;
        setDate(new Date(cellDate));
        setMonthDate(new Date(cellDate.getFullYear(), cellDate.getMonth(), 1));
      }}
      disabled={!isClickable}
      className={`w-11 h-11 rounded-full flex items-center justify-center m-[2px] transition-all duration-100 ${
        isClickable ? "cursor-pointer" : "opacity-25 cursor-not-allowed"
      }`}
      style={{
        background: isSelected ? PRIMARY : "transparent",
        color: isSelected
          ? "#fff"
          : isCurrentMonth
          ? isDark
            ? "#d2d2d2"
            : "#111827"
          : isDark
          ? "#666"
          : "#aaa",
        border: "none",
      }}
    >
      <div className="text-sm pointer-events-none">{cellDate.getDate()}</div>
    </button>
  );
})}

          </div>
        </div>

        {/* slots */}
        <div className="mb-4">
          <h3 className="mb-2 text-sm">Wybierz godzinƒô</h3>
          {loadingSlots ? (<div className={`${subtextClass}`}>≈Åadowanie termin√≥w‚Ä¶</div>) : slots.length ? (
            <div className="flex gap-2 overflow-x-auto pb-2 hide-scrollbar">
              {slots.map((s, i) => {
                const empColor = PRIMARY;
                const isSelected = selectedSlot && selectedSlot.employee_id === s.employee_id && selectedSlot.start_time === s.start_time;
                return (
                  <div key={i} className="flex-shrink-0">
                    <button onPointerDown={() => handleSlotClick(s)} onTouchStart={(e) => e.stopPropagation()} className={`flex flex-col items-center justify-center py-2 px-3 rounded-[22px] min-w-[76px] min-h-[54px] text-center transition-transform duration-100`} style={{ background: isSelected ? empColor : (isDark ? "#0f0f10" : "#ffffff"), color: isSelected ? "#fff" : (isDark ? "#fff" : "#111827"), border: isSelected ? `2px solid ${empColor}` : `1px solid ${isDark ? "#3d3c3c" : "#e5e7eb"}`, boxShadow: isSelected ? `0 6px 18px ${empColor}33` : "none", touchAction: "manipulation" }}>
                      <div className="text-sm font-extrabold leading-none mb-0.5 select-none pointer-events-none">{s.start_time}</div>
                      <div className={`text-xs leading-none pointer-events-none ${isSelected ? "" : subtextClass}`}>{shortName(s.employee_name)}</div>
                    </button>
                  </div>
                );
              })}
            </div>
          ) : (<div className={`${subtextClass}`}>{msg || "Brak dostƒôpnych termin√≥w"}</div>)}
        </div>
      </div>

      {/* bottom summary */}
      <div className={`rounded-xl ${isDark ? "bg-gray-800" : "bg-white"} p-4 mt-2 mb-7 max-w-[1100px] mx-auto border ${borderClass}`}>
        <div className="flex justify-between items-center gap-3">
          <div className="min-w-0">
            <div className="text-base font-semibold truncate">{service?.name || "‚Äî"}
              <div className="mt-1">
                <button onClick={() => setShowServiceSelect(!showServiceSelect)} className={`text-xs underline ${isDark ? "text-gray-400 hover:text-gray-200" : "text-gray-600 hover:text-gray-800"}`}>Zmie≈Ñ us≈Çugƒô</button>
              </div>
              {showServiceSelect && (
                <div className={`mt-2 border rounded-lg p-2 ${isDark ? "bg-gray-900 border-gray-700" : "bg-gray-50 border-gray-300"}`}>
                  {availableServices.length === 0 ? (<div className={`text-sm ${subtextClass}`}>Brak innych us≈Çug</div>) : (
                  <select
  onChange={(e) => handleChangeService(e.target.value)}
  value={service?.id || ""}
  className={`w-full p-2 rounded-md text-sm ${isDark ? "bg-gray-800 text-white" : "bg-white text-gray-900"}`}
>
  <option value="">Wybierz us≈Çugƒô</option>

  {availableServices
  .filter((svc) => {
    if (!employees || employees.length === 0) return true;

    // üîπ Je≈õli wybrano konkretnego pracownika
    if (selectedEmployee !== "any") {
      return svc.employee_ids?.includes(selectedEmployee);
    }

    // üîπ Je≈õli wybrano "Dowolna" ‚Üí pokazuj us≈Çugi przypisane do kogokolwiek z aktualnych pracownik√≥w
    const currentEmpIds = employees.map((e) => e.id);
    return svc.employee_ids?.some((id) => currentEmpIds.includes(id));
  })
  .map((s) => (
    <option key={s.id} value={s.id}>
      {s.name} ‚Äî {formatPrice(s.price)} z≈Ç
    </option>
  ))}

</select>


                  )}
                </div>
              )}
            </div>

            <div className={`mt-1 text-sm ${subtextClass}`}>{selectedSlot ? (selectedSlot.employee_name?.split(" ")[0] || selectedSlot.employee_name) : selectedEmployee === "any" ? "Dowolna osoba" : (employees.find(e => e.id === selectedEmployee)?.name?.split(" ")[0] || "‚Äî")}</div>

            <div className="mt-3">
              <button onPointerDown={() => setAddonsOpen(!addonsOpen)} onTouchStart={(e) => e.stopPropagation()} className={`inline-flex items-center gap-2 px-4 py-2 rounded-full text-sm border ${borderClass} ${isDark ? "bg-gray-800" : "bg-white"} ${isDark ? "text-white" : "text-gray-900"} cursor-pointer`}><Plus size={14} /><span className="pointer-events-none">Dodatki{selectedAddons.length ? ` (${selectedAddons.length})` : ""}</span></button>
              {addonsOpen && (
                <div className={`mt-2 p-2 rounded-md ${isDark ? "bg-gray-900 border-gray-800" : "bg-gray-50 border-gray-200"} border max-w-[520px]`}>
                  {addons.length === 0 ? (<div className={`${subtextClass}`}>Brak dodatk√≥w</div>) : (
                    <div className="grid gap-2">{addons.map((a) => {
                      const selected = selectedAddons.some((x) => x.id === a.id);
                      return (
                        <div key={a.id} onPointerDown={() => toggleAddon(a)} className={`flex justify-between items-center p-2 rounded-md cursor-pointer ${selected ? (isDark ? "bg-[#2b1a0f]" : "bg-[rgba(229,123,44,0.08)]") : (isDark ? "bg-gray-800" : "bg-white")}`}>
                          <div>
                            <div className={`text-sm ${isDark ? "text-white" : "text-gray-900"}`}>{a.name}</div>
                            <div className={`text-xs ${subtextClass}`}>+{formatPrice(a.price)} z≈Ç ‚Ä¢ +{a.duration_minutes} min</div>
                          </div>
                          <div>{selected ? <Check size={18} color={PRIMARY} /> : <div style={{ width: 18, height: 18 }} />}</div>
                        </div>
                      );
                    })}</div>
                  )}
                </div>
              )}
            </div>
          </div>

          <div className="text-right min-w-[120px]">
            <div className="text-lg font-extrabold">{formatPrice(totalPrice)} z≈Ç</div>
            <div className={`mt-1 text-sm ${subtextClass}`}>{selectedSlot ? `${selectedSlot.start_time} ‚Äì ${computedEndTime || selectedSlot.end_time}` : `${totalDuration} min`}</div>
          </div>
        </div>

        <div className="mt-3">
          <button onClick={() => { if (!selectedSlot) return; setShowSummaryModal(true); }} disabled={!selectedSlot} className={`w-full rounded-md py-3 text-base font-semibold ${selectedSlot ? "" : "opacity-60 cursor-not-allowed"}`} style={{ background: selectedSlot ? PRIMARY : (isDark ? "#333" : "#cbd5e1"), color: "#fff", border: "none" }}>Dalej</button>
        </div>
      </div>

      {/* confirmation modal */}
      {showSummaryModal && (
        <div className={`fixed inset-0 z-[9999] flex items-center justify-center p-6 ${modalBg}`}>
          {msg.includes("‚úÖ") ? (
            <div className={`w-full max-w-[480px] text-center p-8 rounded-2xl ${isDark ? "bg-gray-800" : "bg-white"} shadow-lg`}>
              <h2 className="text-[26px] font-bold" style={{ color: PRIMARY }}>üéâ Gratulacje!</h2>
              <p className={`mt-4 text-base ${subtextClass}`}>Twoja wizyta zosta≈Ça um√≥wiona na <strong>{prettyDate(date)}, godz. {selectedSlot?.start_time}</strong>.</p>
              <p className={`mt-2 text-sm ${subtextClass}`}>Dziƒôkujemy za skorzystanie z naszych us≈Çug!</p>
              <button onPointerDown={() => { setMsg(""); setShowSummaryModal(false); window.location.href = "/salons"; }} onTouchStart={(e) => e.stopPropagation()} className="mt-6 w-full py-3 rounded-md font-semibold" style={{ background: PRIMARY, color: "#fff" }}>Zamknij</button>
            </div>
          ) : (
            <div className={`w-full max-w-[480px] p-6 rounded-2xl ${isDark ? "bg-gray-800" : "bg-white"} shadow-lg`}>
              <div className="flex items-center mb-4">
                <button onPointerDown={() => setShowSummaryModal(false)} className={`mr-3 p-1 ${isDark ? "text-white" : "text-gray-900"}`}>‚Üê</button>
                <h2 className="text-xl font-bold m-0">Sprawd≈∫ i potwierd≈∫</h2>
              </div>
              <div className={`text-sm ${subtextClass} mb-2`}>{prettyDate(date)} ¬∑ {selectedSlot?.start_time}</div>

              {/* Salon info */}
              {(() => {
                try {
                  const salon = JSON.parse(localStorage.getItem("selectedSalon"));
                  if (salon) return <div className={`text-sm ${subtextClass} mb-4`}>{salon.name} ¬∑ {salon.street} {salon.street_number}, {salon.city}</div>;
                } catch {}
                return null;
              })()}

              <div className={`p-4 rounded-lg mb-4 border ${borderClass} ${isDark ? "bg-gray-800" : "bg-white"}`}>
                <div className="flex justify-between items-start">
                  <div>
                    <div className="text-base font-semibold">{service?.name || "Us≈Çuga"}</div>
                    <div className={`text-sm mt-1 ${subtextClass}`}>Pracownik: {selectedSlot ? (selectedSlot.employee_name?.split(" ")[0] || selectedSlot.employee_name) : selectedEmployee === "any" ? "Dowolna osoba" : (employees.find((e) => e.id === selectedEmployee)?.name?.split(" ")[0] || "‚Äî")}</div>
                    {selectedAddons.length > 0 && (
                      <div className="mt-3">
                        <div className={`text-sm ${subtextClass} mb-1`}>Dodatki:</div>
                        <ul className="list-disc pl-5 text-sm" style={{ color: isDark ? "#fff" : "#111827" }}>{selectedAddons.map((a) => <li key={a.id} className="mb-1">{a.name} (+{formatPrice(a.price)} z≈Ç, +{a.duration_minutes} min)</li>)}</ul>
                      </div>
                    )}
                  </div>

                  <div className="text-right">
                    <div className="text-base font-semibold">{formatPrice(totalPrice)} z≈Ç</div>
                    <div className={`text-sm mt-1 ${subtextClass}`}>{selectedSlot?.start_time} ‚Äì {computedEndTime}</div>
                  </div>
                </div>

                <hr className={`my-3 border-t ${borderClass}`} />
                <div className="flex justify-between text-base"><span>Suma</span><strong>{formatPrice(totalPrice)} z≈Ç</strong></div>
              </div>

              <div className="text-center mb-4"><div className={`text-sm ${subtextClass}`}>Do zap≈Çaty na miejscu: <strong className={`${isDark ? "text-white" : "text-gray-900"}`}>{formatPrice(totalPrice)} z≈Ç</strong></div></div>

              <button onClick={confirmBooking} className="w-full py-3 rounded-md font-semibold" style={{ background: PRIMARY, color: "#fff" }}>Potwierd≈∫ i um√≥w</button>

              {msg && !msg.includes("‚úÖ") && (<div className="mt-4 rounded-md p-2 text-center" style={{ background: successBg, color: isDark ? "#ff7676" : "#9b1f1f" }}>{msg}</div>)}
            </div>
          )}
        </div>
      )}

      {msg && (<div className="fixed left-5 bottom-24 rounded-md px-3 py-2" style={{ background: successBg, color: isDark ? "#fff" : "#111827" }}>{msg}</div>)}
    </div>
  );
}
